Lyapunov exponents

Date dim. 03 octobre 2010 Tags Lyapunov / Lorenz / bifurcation / chaos / matplotlib / python
\begin{align*} f(x) = \sqrt{\frac{1}{\sqrt{2}}+\sqrt{x^2+1}} \\ \lim \limits_{x \to +\infty} (x^2 + 3x + 1) = +\infty \end{align*} 
#!/usr/bin/env python3

"""
Programme pour calculer les exposants de Lyapunov
du système de Lorenz
Source : Dang-Vu et Delcarte
#http://sprott.physics.wisc.edu/chaos/wolfflow.bas
"""

# appel des bibliothèques scientifiques
import numpy as np
from scipy.integrate import odeint
from math import log as ln
from math import sqrt
import math, operator, random

# Produit scalaire et norme
def ps(u,v):
  u1, u2, u3 = u
  v1, v2, v3 = v
  return float(u1*v1 + u2*v2 + u3*v3)

def norme(u):return sqrt(ps(u,u))

# Système de Lorenz
s, r, b = 16, 45.92, 4
def L(V, t):
  x, y, z, v11, v12, v13, v21, v22, v23, v31, v32, v33 = V
  dx = s*(y-x)
  dy = x*(r-z)-y
  dz = x*y-b*z
  dv11 = -s * v11 + s * v12
  dv12 = (r-z)*v11 - v12 -x*v13
  dv13 = y*v11 + x*v12 - b*v13
  dv21 = -s * v21 + s * v22
  dv22 = (r-z)*v21 - v22 -x*v23
  dv23 = y*v21 + x*v22 - b*v23
  dv31 = -s * v31 + s * v32
  dv32 = (r-z)*v31 - v32 -x*v33
  dv33 = y*v31 + x*v32 - b*v33
  return [dx, dy, dz, dv11, dv12, dv13, dv21, dv22, dv23, dv31, dv32, dv33]

def l(X, t):
  x, y, z = X
  dx = s*(y-x)
  dy = x*(r-z)-y
  dz = x*y-b*z
  return [dx, dy, dz]

def Jacobi(x, y, z):
  return np.matrix([[-s ,  s,  0],
                    [r-z, -1, -x],
                    [y  ,  x, -b]])

X0 = [10, 1, 0]
dt = 0.001
T = 20
N = int(T/dt)
orbit = odeint(l, X0, np.arange(0, 5, dt))
X0 = orbit[-1]

print('(1) First method')
print('Beginning calculating',end='')

v1 = np.matrix([1, 0, 0]).T
v2 = np.matrix([0, 1, 0]).T
v3 = np.matrix([0, 0, 1]).T

orbit = odeint(l, X0, np.arange(0, T, dt))

LE1 = 0
LE2 = 0
LE3 = 0

for k in range(N):
  if k%100==0:print('.',end='')
  x, y, z = orbit[k][0], orbit[k][1], orbit[k][2]
  V0 = x, y, z, float(v1[0]), float(v1[1]), float(v1[2]), float(v2[0]), float(v2[1]), float(v2[2]),float(v3[0]), float(v3[1]), float(v3[2])
  V = odeint(L, V0, [0, dt])
  v1 = np.matrix([V[-1][3], V[-1][4], V[-1][5]]).T
  v2 = np.matrix([V[-1][6], V[-1][7], V[-1][8]]).T
  v3 = np.matrix([V[-1][9], V[-1][10], V[-1][11]]).T
  LE1 = LE1 + ln(norme(v1))
  LE2 = LE2 + ln(norme(v2))
  LE3 = LE3 + ln(norme(v3))
  v1 = v1/norme(v1)
  v2 = (v2 - ps(v2,v1)*v1)/norme(v2 - ps(v2,v1)*v1)
  v3 = (v3 - ps(v3,v2)*v2 - ps(v3,v1)*v1)/norme(v3 - ps(v3,v2)*v2 - ps(v3,v1)*v1)

print('\n')
LE1, LE2, LE3 = LE1/T, LE2/T, LE3/T
print('Lyapunov exponents :',LE1, LE2, LE3)
print('Trace :', LE1+LE2+LE3)

print('\n')
print('(2) Second method')
print('Beginning calculating',end='')

v1 = np.matrix([1, 0, 0]).T
v2 = np.matrix([0, 1, 0]).T
v3 = np.matrix([0, 0, 1]).T

orbit = odeint(l, X0, np.arange(0, T, dt))

LE1 = 0
LE2 = 0
LE3 = 0

for k in range(N):
  if k%100==0:print('.',end='')
  x, y, z = orbit[k][0], orbit[k][1], orbit[k][2]
  A = np.eye(3) + Jacobi(x, y, z)*dt
  v1 = A*v1
  v2 = A*v2
  v3 = A*v3
  LE1 = LE1 + ln(norme(v1))
  LE2 = LE2 + ln(norme(v2))
  LE3 = LE3 + ln(norme(v3))
  v1 = v1/norme(v1)
  v2 = (v2 - ps(v2,v1)*v1)/norme(v2 - ps(v2,v1)*v1)
  v3 = (v3 - ps(v3,v2)*v2 - ps(v3,v1)*v1)/norme(v3 - ps(v3,v2)*v2 - ps(v3,v1)*v1)

print('\n')
LE1, LE2, LE3 = LE1/T, LE2/T, LE3/T
print('Lyapunov exponents :',LE1, LE2, LE3)
print('Trace :', LE1+LE2+LE3)
