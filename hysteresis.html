Hysteresis

Date ven. 01 janvier 2016

The following program, written in python , gives a numerical computation of hysteresis bifurcation in the dynamical system

\begin{equation*} x' = \lambda + x - x^3 \end{equation*}

with parameter \(\lambda\) [Hale et al. (1996)].

Numerical computation of hysteresis


#!/usr/bin/env python3

"""
Numerical computation of the bifurcation in system
x' = lambda + x - x^3
"""


# Scientific libraries

from matplotlib import pyplot as plt
import numpy as np
from scipy.integrate import odeint
from scipy.optimize import fsolve

# Step 1

lambda0 = -5
def f(x):
    return lambda0 + x - x**3
f0 = 1
x0 = fsolve(f, f0)[0]


# Step 2

def system(X, t):
    l, x = X
    dl = -(1-3*x**2)
    dx = 1
    return [dl, dx]

CI = [lambda0, x0]

time = np.arange(0, 10, 0.1)
result1 = odeint(system, CI, time)
result2 = odeint(system, CI, -time)

l1, x1 = result1.T
l2, x2 = result2.T

# Figure

fig, ax = plt.subplots()
ax.plot(l1, x1, 'r')
ax.plot(l2, x2, 'r')
ax.set_xlim(-5,5)
ax.set_ylim(-3,3)
ax.set_xlabel(r"$\lambda$", fontsize=20)
ax.set_ylabel(r"$\bar{x}$", fontsize=20, rotation='horizontal')

plt.show()
